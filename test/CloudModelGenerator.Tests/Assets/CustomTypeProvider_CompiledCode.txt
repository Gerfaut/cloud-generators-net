using System;
using KenticoCloud.Delivery;

namespace KenticoCloudModels
{
    public class CustomTypeProvider : ICodeFirstTypeProvider
    {
        public Type GetType(string contentType)
        {
            switch (contentType)
            {
                case "article":
                    return typeof(Article);
                case "office":
                    return typeof(Office);
                default:
                    return null;
            }
        }
    }
}
